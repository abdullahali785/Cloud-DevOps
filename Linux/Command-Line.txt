1) Navigation

ls # List files/folders
ls-l # Detailed List
ls-a # Includes hidden files

cd # Change Directory
cd path/to/folders
cd ~ # Home
cd .. # One folder up

pwd # Shows current directory


2) File Operations

cp # Copy files
cp file1.txt file2.txt # Copies file1 into file2
cp -r folder1 folder2 # Recursive copy 

mv # Move/rename files
mv old.txt new.txt
mv file.txt /path/to/destination/

rm # Remove/delete files
rm file.txt
rm -r folder

mkdir # New directory
touch # New file
